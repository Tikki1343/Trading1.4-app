const CACHE_NAME = "trading-app-v1"; const urlsToCache = ["/", "/index.html", "/static/js/main.chunk.js", "/static/js/bundle.js"]; self.addEventListener("install", (event) => { event.waitUntil( caches.open(CACHE_NAME).then((cache) => cache.addAll(urlsToCache)) ); }); self.addEventListener("fetch", (event) => { event.respondWith( caches.match(event.request).then((response) => { if (response) return response; return fetch(event.request).catch(() => caches.match("/index.html")); }) ); }); self.addEventListener("activate", (event) => { event.waitUntil( caches.keys().then((cacheNames) => Promise.all( cacheNames .filter((name) => name !== CACHE_NAME) .map((name) => caches.delete(name)) ) ) ); });
